def fibonacci(n):
if n <= 1:
return n
return fibonacci(n - 1) + fibonacci(n - 2)

import math

def softmax(x):
exp_x = [math.exp(v) for v in x]
total = sum(exp_x)
return [v / total for v in exp_x]

def merge(indices, pair, new_index):
out = []
i = 0
while i < len(indices):
if i + 1 < len(indices) and (indices[i], indices[i + 1]) == pair:
out.append(new_index)
i += 2
else:
out.append(indices[i])
i += 1
return out
